# مرحله: ساخت ماتریس بروزرسانی (Incidence Matrix) رابطه برتری
# منبع: بخش ۵.۲ کتاب Hyperparameter Tuning with R
# هدف: مقایسه میانگین عملکرد الگوریتم‌ها برای تعیین A_i ⪰ A_j
# =============================================================================

# ایجاد ماتریس صفر با ابعاد تعداد الگوریتم‌ها × تعداد الگوریتم‌ها
I <- matrix(0, nrow = nrow(Y), ncol = nrow(Y))

# تنظیم نام سطرها و ستون‌ها بر اساس نام الگوریتم‌ها (A_1 تا A_5)
rownames(I) <- colnames(I) <- rownames(Y)

# حلقه دوگانه برای مقایسه میانگین هر جفت الگوریتم
for (i in 1:nrow(Y)) {
  for (j in 1:nrow(Y)) {
    # اگر میانگین عملکرد الگوریتم i بهتر یا مساوی j باشد → ۱ (A_i ⪰ A_j)
    I[i, j] <- mean(Y[i, ]) >= mean(Y[j, ])
  }
}

# نمایش ماتریس نهایی
I
